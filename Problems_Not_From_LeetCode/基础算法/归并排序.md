#include <iostream>
void Merge(int *A,int *L, int leftCount,int *R, int rightCount)
{
    int leftPre = 0 ;
    int rightPre=0;
    int resPre=0;
    while(leftPre<leftCount&&rightPre<rightCount)
        A[resPre++] = L[leftPre]<R[rightPre]? L[leftPre++]:R[rightPre++];
//    {

//        else

//            A[resPre++] = R[rightPre++];
//        }
//    }
    while(leftPre<leftCount)
        A[resPre++] = L[leftPre++];
    while(rightPre<rightCount)
        A[resPre++] = R[rightPre++];
}
void MergeSort(int *A,int n)
{
    int mid,i,*L,*R;
    if(n<2) return;
    mid = n/2;
    L = new int[mid];
    R = new int[n-mid];
    for(i = 0;i<mid;i++)L[i] = A[i];
    for(i = mid;i < n;i++)R[i-mid]=A[i];

    MergeSort(L,mid);
    MergeSort(R,n-mid);
    Merge(A,L,mid,R,n-mid);
    delete[]R;
    delete[]L;
}
int main() {
    int n;
    std::cin>>n;
    int* arr = new int[n];
    for(int i =0;i<n;i++)
    {
        std::cin>>arr[i];
    }
    int k;
    std::cin>>k;
    MergeSort(arr,n);
    std::cout<<arr[n-k]<<std::endl;
    return 0;
}
